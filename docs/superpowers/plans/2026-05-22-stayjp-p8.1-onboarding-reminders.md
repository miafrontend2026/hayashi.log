```ts
// src/lib/notifications.ts
import * as Notifications from "expo-notifications";
import { Platform } from "react-native";

export type ReminderTime = "off" | "morning" | "noon" | "evening" | "night";

const TIME_MAP: Record<Exclude<ReminderTime, "off">, { hour: number; minute: number; label: string }> = {
  morning: { hour: 8, minute: 0, label: "早上 8:00" },
  noon:    { hour: 12, minute: 30, label: "中午 12:30" },
  evening: { hour: 19, minute: 0, label: "晚上 7:00" },
  night:   { hour: 21, minute: 30, label: "夜晚 9:30" },
};

export function timeLabel(time: ReminderTime): string {
  if (time === "off") return "不提醒";
  return TIME_MAP[time].label;
}

const REMINDER_ID = "stayjp-daily-reminder";

/** Request permission. Returns true if granted. */
export async function requestPermission(): Promise<boolean> {
  if (Platform.OS === "web") return false;
  const { status: existing } = await Notifications.getPermissionsAsync();
  if (existing === "granted") return true;
  const { status } = await Notifications.requestPermissionsAsync();
  return status === "granted";
}

/** Cancel any prior reminder. */
export async function cancelReminder(): Promise<void> {
  if (Platform.OS === "web") return;
  try {
    await Notifications.cancelScheduledNotificationAsync(REMINDER_ID);
  } catch { /* ignore */ }
}

/** Schedule a daily local notification at the chosen slot. No-op for "off". */
export async function scheduleReminder(time: ReminderTime): Promise<void> {
  if (Platform.OS === "web") return;
  await cancelReminder();
  if (time === "off") return;
  const { hour, minute } = TIME_MAP[time];
  await Notifications.scheduleNotificationAsync({
    identifier: REMINDER_ID,
    content: {
      title: "日本再留計劃",
      body: "今日的單字還沒背完,快來複習吧",
      sound: false,
    },
    trigger: {
      type: Notifications.SchedulableTriggerInputTypes.DAILY,
      hour,
      minute,
    } as Notifications.NotificationTriggerInput,
  });
}
```

```tsx
import { scheduleReminder, requestPermission, type ReminderTime } from "../src/lib/notifications";
const REMINDER_OPTS = [
  { v: "off",     label: "不提醒" },
  { v: "morning", label: "早上 8:00" },
  { v: "noon",    label: "中午 12:30" },
  { v: "evening", label: "晚上 7:00" },
  { v: "night",   label: "夜晚 9:30" },
] as const;

const [reminder, setReminder] = useState<ReminderTime>(() => getPref("reminder_time") ?? "off");
async function pickReminder(v: ReminderTime) {
  setReminder(v);
  setPref("reminder_time", v);
  if (v !== "off") {
    const ok = await requestPermission();
    if (!ok) { setReminder("off"); setPref("reminder_time", "off"); return; }
  }
  await scheduleReminder(v);
}

// JSX (below theme block):
<View style={{ marginTop: 24 }}>
  <Text style={{ color: colors.text, fontSize: 14, fontWeight: "600", marginBottom: 10 }}>每日提醒</Text>
  <View style={{ flexDirection: "row", flexWrap: "wrap", gap: 8 }}>
    {REMINDER_OPTS.map((o) => {
      const on = reminder === o.v;
      return (
        <Pressable key={o.v} onPress={() => pickReminder(o.v)}
          className="px-4 py-2 rounded-full"
          style={{ backgroundColor: on ? colors.primary : colors.chip }}>
          <Text style={{ color: on ? "white" : colors.text, fontSize: 13, fontWeight: on ? "600" : "400" }}>
            {o.label}
          </Text>
        </Pressable>
      );
    })}
  </View>
</View>
```
