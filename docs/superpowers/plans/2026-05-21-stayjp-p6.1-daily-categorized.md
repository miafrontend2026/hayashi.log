# StayJP P6.1 — Daily Mode + Categorized Sections Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development. Steps use `- [ ]`.

**Goal:** Match the web's vocab/grammar browse experience. P6 dumped all 800 words at once — wrong. Web's pattern (`index.html` lines 2338-2738):
- **Daily mode (default ON):** show only today's 20-word batch (vocab) / 10-point batch (grammar)
- **Progress bar:** persistent per-level offset in `daily_progress` localStorage / Firestore field
- **Batch navigation:** 前一批 / 後一批 / 重置 buttons
- **Category sections:** grouped by 名詞 / 動詞 / い形 / な形 / 副詞 / 他 with section headers + counts
- **跟讀本批 button:** shadow only the current batch
- **Show all toggle:** switch from daily mode to full list (web's "全部"按鈕)

**Architecture:**
- New shared `src/stores/dailyProgressStore.ts` syncs the `daily_progress` field to/from web via existing `webSync` helpers.
- Vocab category constants ported as `src/vocab/cats.ts`. Grammar groups by free-form `cat` field (no fixed list).
- `VocabList` + `GrammarList` get refactored to render `SectionList` (RN's grouped FlatList) when in full mode + plain FlatList in daily mode.
- Each list gets a header block: progress + daily-mode toggle + batch nav + 跟讀本批 button.
- 跟讀本批 wires through the existing `useLaunchIntent` store with a new queue-payload pattern OR a direct `useShadow.start(queue)` call (cleaner).

**Web ref:**
- Constants & helpers: `index.html` 2338-2426 (daily progress fns) and 2633-2638 (vocab cats)
- Render logic: `index.html` 2648-2780 (renderVocab) and ~2780-2950 (renderGrammar — similar pattern)
- Web sync field: `daily_progress` in `SYNC_KEYS` at line 1489

**Spec ref:** `stay-jp-notes/docs/superpowers/specs/2026-05-19-stayjp-app-design.md` § 5.B and § 5.C.

## Web schema for `daily_progress`

```ts
type DailyProgress = Record<Level, {
  date: string;       // ISO yyyy-mm-dd
  learned: number;    // unused in offset tracking; kept for forward compat
  offset: number;     // unused; same as totalOffset on web
  totalOffset: number; // current batch start
}>;
```

Web's merge in `loadCloudData`: per-key object merge — local newer (by `lastReview` for SRS); for daily_progress, web does basic `Object.entries` merge keeping cloud values when local missing. We mirror: prefer the entry with the most recent `date`. If dates equal, take the larger `totalOffset`.

// src/vocab/groupByCategory.ts
import type { Word } from "../types/content";
import { VOCAB_CATS, type VocabCat } from "./cats";

export interface VocabSection {
  cat: VocabCat;
  data: Word[];
}

export function groupVocabByCategory(words: Word[]): VocabSection[] {
  const buckets: Record<string, Word[]> = {};
  VOCAB_CATS.forEach(c => { buckets[c.key] = []; });
  for (const w of words) {
    const cat = VOCAB_CATS.find(c => c.key === w.c);
    if (cat) buckets[cat.key].push(w);
    else buckets["他"].push(w);
  }
  return VOCAB_CATS
    .map(cat => ({ cat, data: buckets[cat.key] }))
    .filter(s => s.data.length > 0);
}
```

// src/grammar/groupByCategory.ts
import type { GrammarItem } from "../types/content";

export interface GrammarSection {
  cat: string;
  data: GrammarItem[];
}

export function groupGrammarByCategory(items: GrammarItem[]): GrammarSection[] {
  const order: string[] = [];
  const buckets: Record<string, GrammarItem[]> = {};
  for (const item of items) {
    const cat = item.cat || "其他";
    if (!buckets[cat]) { buckets[cat] = []; order.push(cat); }
    buckets[cat].push(item);
  }
  return order.map(cat => ({ cat, data: buckets[cat] }));
}

export interface DailyProgressEntry {
  date: string;
  learned: number;
  offset: number;
  totalOffset: number;
}

export type DailyProgressMap = Partial<Record<"n5"|"n4"|"n3"|"n2"|"n1", DailyProgressEntry>>;

export interface WebUserDoc {
  // ...existing fields...
  daily_progress?: DailyProgressMap;
  daily_progress_grammar?: DailyProgressMap;  // mirror — web uses grammar-specific key
}

/** Merge: per-level pick the entry with the newer `date`; if equal, larger totalOffset. */
export function mergeDailyProgress(
  local: DailyProgressMap,
  cloud: DailyProgressMap,
): DailyProgressMap {
  const out: DailyProgressMap = { ...cloud };
  (Object.keys(local) as Array<keyof DailyProgressMap>).forEach((lv) => {
    const lv2 = lv as keyof DailyProgressMap;
    const l = local[lv2]!;
    const c = out[lv2];
    if (!c) { out[lv2] = l; return; }
    if (l.date > c.date) { out[lv2] = l; return; }
    if (l.date === c.date && l.totalOffset > c.totalOffset) out[lv2] = l;
  });
  return out;
}
```

```ts
// src/stores/dailyProgressStore.ts
import { getJSON, setJSON } from "../lib/storage";
import { schedulePushField, type DailyProgressEntry, type DailyProgressMap } from "../lib/webSync";
import type { Level } from "../types/content";

const VOCAB_KEY = "daily_progress";
const GRAMMAR_KEY = "daily_progress_grammar";

function todayIso(): string {
  return new Date().toISOString().split("T")[0];
}

function read(key: string): DailyProgressMap {
  return getJSON<DailyProgressMap>(key) ?? {};
}

function write(key: string, value: DailyProgressMap, field: "daily_progress"|"daily_progress_grammar"): void {
  setJSON(key, value);
  schedulePushField(field, value);
}

export function getVocabProgress(level: Level): DailyProgressEntry {
  const all = read(VOCAB_KEY);
  const today = todayIso();
  const cur = all[level];
  if (!cur || cur.date !== today) {
    // Roll forward: keep totalOffset, refresh date
    return { date: today, learned: 0, offset: cur?.totalOffset ?? 0, totalOffset: cur?.totalOffset ?? 0 };
  }
  return cur;
}

export function setVocabProgress(level: Level, prog: DailyProgressEntry): void {
  const all = read(VOCAB_KEY);
  all[level] = prog;
  write(VOCAB_KEY, all, "daily_progress");
}

export function getGrammarProgress(level: Level): DailyProgressEntry {
  const all = read(GRAMMAR_KEY);
  const today = todayIso();
  const cur = all[level];
  if (!cur || cur.date !== today) {
    return { date: today, learned: 0, offset: cur?.totalOffset ?? 0, totalOffset: cur?.totalOffset ?? 0 };
  }
  return cur;
}

export function setGrammarProgress(level: Level, prog: DailyProgressEntry): void {
  const all = read(GRAMMAR_KEY);
  all[level] = prog;
  write(GRAMMAR_KEY, all, "daily_progress_grammar");
}
```

// src/components/BatchNav.tsx
import { View, Text, Pressable } from "react-native";
import { Ionicons } from "@expo/vector-icons";

interface Props {
  level: string;          // upper-cased
  total: number;          // total items in level
  offset: number;         // current batch start
  batchSize: number;      // 20 or 10
  dailyMode: boolean;
  onPrev: () => void;
  onNext: () => void;
  onReset: () => void;
  onToggle: () => void;
  onShadowBatch: () => void;
  /** Label for the action button at the right ("跟讀本批" for vocab, etc.) */
  shadowLabel?: string;
}

export function BatchNav({
  level, total, offset, batchSize, dailyMode,
  onPrev, onNext, onReset, onToggle, onShadowBatch, shadowLabel = "跟讀本批",
}: Props) {
  const start = offset;
  const end = Math.min(offset + batchSize, total);
  const pct = total > 0 ? Math.round((offset / total) * 100) : 0;

  return (
    <View className="bg-white px-5 py-3 border-b border-slate-100">
      <View className="flex-row items-center justify-between mb-2">
        <Text className="text-xs text-slate-500 uppercase tracking-widest font-semibold">{level} 進度</Text>
        <Text className="text-xs text-slate-500">{offset} / {total} ({pct}%)</Text>
      </View>
      <View className="h-1.5 bg-slate-200 rounded-full overflow-hidden">
        <View className="h-full bg-indigo-500" style={{ width: `${pct}%` }} />
      </View>
      <View className="flex-row items-center gap-2 mt-3">
        <Pressable onPress={onToggle}
          className={`flex-1 flex-row items-center justify-center gap-1 py-2 rounded-full ${dailyMode ? "bg-indigo-600" : "bg-slate-100"}`}>
          <Ionicons name="calendar-outline" size={14} color={dailyMode ? "#fff" : "#475569"} />
          <Text className={dailyMode ? "text-white text-sm font-semibold" : "text-slate-700 text-sm"}>
            {dailyMode ? `今日 ${start+1}-${end}` : "顯示全部"}
          </Text>
        </Pressable>
        {dailyMode && (
          <>
            {offset > 0 && (
              <Pressable onPress={onPrev} hitSlop={8}
                className="px-3 py-2 rounded-full bg-slate-100 active:bg-slate-200">
                <Ionicons name="chevron-back" size={16} color="#475569" />
              </Pressable>
            )}
            {end < total && (
              <Pressable onPress={onNext} hitSlop={8}
                className="px-3 py-2 rounded-full bg-slate-100 active:bg-slate-200">
                <Ionicons name="chevron-forward" size={16} color="#475569" />
              </Pressable>
            )}
            {offset > 0 && (
              <Pressable onPress={onReset} hitSlop={8}
                className="px-3 py-2 rounded-full bg-slate-100 active:bg-slate-200">
                <Ionicons name="refresh" size={16} color="#475569" />
              </Pressable>
            )}
          </>
        )}
        <Pressable onPress={onShadowBatch}
          className="px-3 py-2 rounded-full bg-rose-50 active:bg-rose-100 flex-row items-center gap-1">
          <Ionicons name="mic-outline" size={14} color="#e11d48" />
          <Text className="text-rose-700 text-sm font-semibold">{shadowLabel}</Text>
        </Pressable>
      </View>
    </View>
  );
}

### Task 5: VocabList — daily mode + sections + BatchNav

**Files:**
- Modify: `src/vocab/VocabList.tsx`

- [ ] **Step 1: Refactor**

Replace VocabList with this structure:
- State: `dailyMode` (default true on first mount per level), `tick` for refresh
- Helpers: get `vocab(level)`, slice by daily offset if daily mode on, group by category
- Render: `<BatchNav>` header + `SectionList` (with section header + VocabRow rows)
- `跟讀本批` button → builds queue from current batch (or full filtered list) and calls `useShadow.start(queue)` + `router.push("/(tabs)/shadowing")`

Implementation skeleton:

```tsx
import { useCallback, useMemo, useState } from "react";
import { SectionList, View, Text, Pressable } from "react-native";
import { useRouter } from "expo-router";
import { allLevels, vocab } from "../lib/content";
import { getSrsData } from "../srs/store";
import { getVocabFavs, toggleVocabFav } from "../shadow/favs";
import { VocabRow } from "./VocabRow";
import { groupVocabByCategory, type VocabSection } from "./groupByCategory";
import { BatchNav } from "../components/BatchNav";
import { getVocabProgress, setVocabProgress } from "../stores/dailyProgressStore";
import { useShadow } from "../shadow/session";
import type { Level, Word } from "../types/content";

const BATCH = 20;

interface Props {
  level: Level;
  onLevelChange: (l: Level) => void;
}

export function VocabList({ level, onLevelChange }: Props) {
  const router = useRouter();
  const [tick, setTick] = useState(0);
  const [dailyMode, setDailyMode] = useState(true);

  const all = useMemo(() => vocab(level), [level]);
  const prog = useMemo(() => getVocabProgress(level), [level, tick]);
  const visible = useMemo(() => {
    if (!dailyMode) return all;
    return all.slice(prog.totalOffset, prog.totalOffset + BATCH);
  }, [all, prog.totalOffset, dailyMode]);

  const srs = useMemo(() => getSrsData(), [tick, level]);
  const favSet = useMemo(() => {
    const s = new Set<string>();
    getVocabFavs().forEach((x) => s.add(`${x.lv ?? ""}::${x.w}`));
    return s;
  }, [tick, level]);

  const sections: VocabSection[] = useMemo(() => groupVocabByCategory(visible), [visible]);

  const onPressWord = useCallback((w: Word) => {
    router.push(`/word/${level}/${encodeURIComponent(w.w)}`);
  }, [router, level]);

  const onToggleFav = useCallback((w: Word) => {
    toggleVocabFav({ j: w.r || w.w, z: "", w: w.w, r: w.r, m: w.m, lv: level });
    setTick((n) => n + 1);
  }, [level]);

  function nextBatch() {
    const newOffset = Math.min(prog.totalOffset + BATCH, all.length);
    setVocabProgress(level, { ...prog, date: new Date().toISOString().split("T")[0], totalOffset: newOffset, offset: newOffset });
    setTick((n) => n + 1);
  }
  function prevBatch() {
    const newOffset = Math.max(prog.totalOffset - BATCH, 0);
    setVocabProgress(level, { ...prog, date: new Date().toISOString().split("T")[0], totalOffset: newOffset, offset: newOffset });
    setTick((n) => n + 1);
  }
  function resetBatch() {
    setVocabProgress(level, { ...prog, date: new Date().toISOString().split("T")[0], totalOffset: 0, offset: 0 });
    setTick((n) => n + 1);
  }
  function shadowThisBatch() {
    const queue = visible.map(w => ({ j: w.r || w.w, z: w.w + (w.m ? ` · ${w.m}` : ""), w: w.w, r: w.r, m: w.m, lv: level }));
    useShadow.getState().start(queue);
    router.push("/(tabs)/shadowing");
  }

  return (
    <View style={{ flex: 1 }}>
      {/* Level pills */}
      <View className="flex-row gap-2 px-5 py-3 bg-white border-b border-slate-100">
        {allLevels.map((lv) => {
          const on = lv === level;
          return (
            <Pressable key={lv} onPress={() => onLevelChange(lv)}
              className={`px-4 py-1.5 rounded-full ${on ? "bg-indigo-600" : "bg-slate-100"}`}>
              <Text className={on ? "text-white font-semibold text-sm" : "text-slate-700 text-sm"}>{lv.toUpperCase()}</Text>
            </Pressable>
          );
        })}
      </View>
      {/* Batch nav */}
      <BatchNav
        level={level.toUpperCase()}
        total={all.length}
        offset={prog.totalOffset}
        batchSize={BATCH}
        dailyMode={dailyMode}
        onPrev={prevBatch}
        onNext={nextBatch}
        onReset={resetBatch}
        onToggle={() => setDailyMode((b) => !b)}
        onShadowBatch={shadowThisBatch}
      />
      <SectionList
        sections={sections.map(s => ({ key: s.cat.key, title: s.cat.name, icon: s.cat.icon, count: s.data.length, data: s.data }))}
        keyExtractor={(item) => `${level}:${item.w}`}
        renderItem={({ item }) => (
          <VocabRow
            word={item}
            level={level}
            srs={srs}
            isFav={favSet.has(`${level}::${item.w}`)}
            onPress={() => onPressWord(item)}
            onToggleFav={() => onToggleFav(item)}
          />
        )}
        renderSectionHeader={({ section }) => (
          <View className="flex-row items-center px-5 py-2 bg-slate-50 border-b border-slate-100">
            <View className="w-6 h-6 rounded bg-indigo-100 items-center justify-center mr-2">
              <Text className="text-indigo-700 font-bold text-xs">{section.icon}</Text>
            </View>
            <Text className="text-sm font-semibold text-slate-800">{section.title}</Text>
            <Text className="ml-auto text-xs text-slate-500">{section.count}</Text>
          </View>
        )}
        stickySectionHeadersEnabled={false}
        initialNumToRender={20}
      />
    </View>
  );
}
```
