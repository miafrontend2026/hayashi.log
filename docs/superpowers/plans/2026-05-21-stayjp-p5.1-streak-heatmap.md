# StayJP P5.1 — Real Streak Heatmap Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development. Steps use `- [ ]`.

**Goal:** Replace the StreakCalendar placeholder with a real 13-week GitHub-style heatmap driven by an actual daily activity log. Compute current + longest streak. Bidirectional sync to web's existing `study_log` Firestore field. Wire activity logging at the natural touchpoints: Flashcard rate + Shadow session-complete + Word detail star toggle.

**Web reference:** `stay-jp-notes/calendar.js` (full module). Key behaviors:
- `study_log` schema: `{ [yyyy-mm-dd]: { vocab, grammar, quiz, minutes } }`
- `logActivity(type)` increments today's bucket
- 91 days (~13 weeks) of heatmap from today backwards
- Intensity 0–3: 0=none, 1=≤5, 2=≤15, 3=>15
- Streak: walk backwards from today counting consecutive non-zero days; longest = scan all dates, find longest consecutive run

**Architecture:**
- New pure `src/study/streak.ts` exports `getStreaks(log)`, `getLevel(dayData)`, `today()`, `summarizeDay(log, day)`. All deterministic, TDD-friendly.
- New `src/study/store.ts` — MMKV-backed log + sync via webSync's schedulePushField (`study_log` field already on `SYNC_KEYS` web side).
- Extend `WebUserDoc` to include `study_log`. Add `mergeStudyLog(local, cloud)` — per-day deep merge taking the **larger** of each counter (web's `loadCloudData` for objects uses `lastReview` tie-break, but study_log doesn't have one; safest = take max counts per day per type to never undercount activity that happened on one device but not the other).
- Wire `logActivity` calls at:
  - `useSession.rate()` in `src/flashcard/session.ts` — log `'vocab'` once per rating
  - `useShadow` state transition to `'done'` — log `'grammar'` if all items had `lv` undefined (grammar examples) OR log `'vocab'` if vocab queue. Detect by checking queue items' `lv` field.
  - `toggleVocabFav` in `src/shadow/favs.ts` — light touch: log `'vocab'` so starring counts as activity too (matches web behavior implicitly)
- Real heatmap: `src/components/StreakCalendar.tsx` rewritten as 13-week grid (7 rows × 13 cols) + month label row + streak header row + today's progress.

```ts
// src/study/streak.ts
import type { DayData, StudyLog } from "./types";

const MS_PER_DAY = 86_400_000;

export function dayKeyAt(d: Date): string {
  return d.toISOString().split("T")[0];
}

export function isActiveDay(day: DayData | undefined): boolean {
  if (!day) return false;
  return (day.vocab > 0) || (day.grammar > 0) || (day.quiz > 0);
}

export function getLevel(day: DayData | undefined): 0 | 1 | 2 | 3 {
  if (!day) return 0;
  const total = (day.vocab || 0) + (day.grammar || 0) + (day.quiz || 0);
  if (total === 0) return 0;
  if (total <= 5) return 1;
  if (total <= 15) return 2;
  return 3;
}

export function getStreaks(log: StudyLog, now: Date = new Date()): { current: number; longest: number } {
  // Current streak: walk backwards from today. If today is empty, skip today
  // and start counting from yesterday (per web behavior).
  let current = 0;
  for (let i = 0; i < 366; i++) {
    const day = new Date(now.getTime() - i * MS_PER_DAY);
    const key = dayKeyAt(day);
    const entry = log[key];
    if (isActiveDay(entry)) {
      current++;
    } else {
      if (i === 0) continue; // today's bucket may be empty yet — keep checking yesterday
      break;
    }
  }

  // Longest streak: scan all keys
  const dates = Object.keys(log).filter((k) => isActiveDay(log[k])).sort();
  let longest = 0;
  let run = 0;
  for (let i = 0; i < dates.length; i++) {
    if (i === 0) { run = 1; }
    else {
      const prev = new Date(dates[i - 1] + "T00:00:00Z");
      const curr = new Date(dates[i] + "T00:00:00Z");
      const diff = (curr.getTime() - prev.getTime()) / MS_PER_DAY;
      run = diff === 1 ? run + 1 : 1;
    }
    if (run > longest) longest = run;
  }

  return { current, longest };
}
```

```ts
import type { StudyLog, DayData } from "../study/types";

export interface WebUserDoc {
  // ...existing
  study_log?: StudyLog;
}

/**
 * Per-day merge. For each day, take the MAX of each counter type across local and cloud.
 * Reason: a user might rate 10 cards on phone offline, then 5 on web; we want total reflected.
 * Safer than last-write-wins for additive counters.
 */
export function mergeStudyLog(local: StudyLog, cloud: StudyLog): StudyLog {
  const out: StudyLog = { ...cloud };
  for (const [day, l] of Object.entries(local)) {
    const c = out[day];
    if (!c) { out[day] = l; continue; }
    out[day] = {
      vocab: Math.max(l.vocab ?? 0, c.vocab ?? 0),
      grammar: Math.max(l.grammar ?? 0, c.grammar ?? 0),
      quiz: Math.max(l.quiz ?? 0, c.quiz ?? 0),
      minutes: Math.max(l.minutes ?? 0, c.minutes ?? 0),
    };
  }
  return out;
}
```

```ts
// src/study/store.ts
import { getJSON, setJSON } from "../lib/storage";
import { schedulePushField } from "../lib/webSync";
import { dayKeyAt } from "./streak";
import type { StudyLog, DayData, ActivityType } from "./types";

const KEY = "study_log";

export function getStudyLog(): StudyLog {
  return getJSON<StudyLog>(KEY) ?? {};
}

export function setStudyLog(log: StudyLog): void {
  setJSON(KEY, log);
  schedulePushField("study_log", log);
}

export function logActivity(type: ActivityType, now: Date = new Date()): void {
  const log = getStudyLog();
  const key = dayKeyAt(now);
  const day: DayData = log[key] ?? { vocab: 0, grammar: 0, quiz: 0 };
  day[type] = (day[type] ?? 0) + 1;
  log[key] = day;
  setStudyLog(log);
}

export function getTodaySummary(now: Date = new Date()): DayData & { total: number } {
  const key = dayKeyAt(now);
  const log = getStudyLog();
  const d = log[key] ?? { vocab: 0, grammar: 0, quiz: 0 };
  return { ...d, total: (d.vocab || 0) + (d.grammar || 0) + (d.quiz || 0) };
}
```

```tsx
// src/components/StreakCalendar.tsx
import { useMemo } from "react";
import { Text, View } from "react-native";
import { Ionicons } from "@expo/vector-icons";
import { StayCard } from "./StayCard";
import { colors } from "../theme";
import { getStreaks, getLevel, dayKeyAt } from "../study/streak";
import { getStudyLog, getTodaySummary } from "../study/store";

const COLS = 13;  // weeks
const ROWS = 7;   // days
const TOTAL = COLS * ROWS;  // 91
const CELL = 16;
const GAP = 3;

const LEVEL_COLORS = [
  "#E5E7EB",  // 0 — empty
  "rgba(212,101,74,0.3)",   // 1 — light
  "rgba(212,101,74,0.6)",   // 2 — medium
  "#D4654A",                 // 3 — full
];

interface Props {
  // Optional override for tests
  now?: Date;
}

export function StreakCalendar({ now: nowProp }: Props = {}) {
  const data = useMemo(() => {
    const now = nowProp ?? new Date();
    const log = getStudyLog();
    const today = getTodaySummary(now);
    const { current, longest } = getStreaks(log, now);

    // Build 91 cells ending at the most recent Saturday (so today is in the last column)
    // Web aligns to Sunday week-start; we use Saturday end which is the equivalent for column layout.
    // Pick: start at today minus 90 days, snap forward to grid based on weekday.
    const endDate = new Date(now);
    endDate.setHours(0, 0, 0, 0);
    const start = new Date(endDate);
    start.setDate(start.getDate() - (TOTAL - 1));

    // Align start to Sunday (column-major reading)
    const startDay = start.getDay();
    start.setDate(start.getDate() - startDay);

    const cells: Array<{ key: string; level: 0|1|2|3; date: Date }> = [];
    const cursor = new Date(start);
    while (cursor <= endDate) {
      const key = dayKeyAt(cursor);
      cells.push({ key, level: getLevel(log[key]), date: new Date(cursor) });
      cursor.setDate(cursor.getDate() + 1);
    }

    // Group by week (column)
    const cols: typeof cells[] = [];
    for (let i = 0; i < cells.length; i += 7) {
      cols.push(cells.slice(i, i + 7));
    }

    // Month labels from cells' first row
    const monthLabels: { col: number; label: string }[] = [];
    let lastMonth = -1;
    cols.forEach((week, ci) => {
      const first = week[0]?.date;
      if (!first) return;
      const m = first.getMonth();
      if (m !== lastMonth) {
        monthLabels.push({ col: ci, label: `${m + 1}月` });
        lastMonth = m;
      }
    });

    const todayStatus = today.total === 0 ? "尚未開始" : (today.total >= 10 ? "完成" : "進行中");

    return { cols, current, longest, today, monthLabels, todayStatus };
  }, [nowProp]);

  return (
    <StayCard className="p-4">
      <View className="flex-row items-center gap-2 mb-3">
        <Ionicons name="flame" size={16} color={colors.primary} />
        <Text style={{ color: colors.primary, fontWeight: "700", fontSize: 14 }}>連續 {data.current} 天</Text>
        <Text style={{ color: colors.textMuted, fontSize: 13 }}>|</Text>
        <Text style={{ color: colors.text, fontSize: 13 }}>最長 {data.longest} 天</Text>
        <Text style={{ color: colors.textMuted, fontSize: 13 }}>|</Text>
        <Text style={{ color: colors.text, fontSize: 13 }}>今日:{data.todayStatus}</Text>
      </View>

      {/* Month labels */}
      <View style={{ flexDirection: "row", paddingLeft: 0, marginBottom: 4 }}>
        {data.cols.map((_, ci) => {
          const lbl = data.monthLabels.find(m => m.col === ci);
          return (
            <View key={ci} style={{ width: CELL + GAP }}>
              {lbl ? <Text style={{ color: colors.textMuted, fontSize: 10 }}>{lbl.label}</Text> : null}
            </View>
          );
        })}
      </View>

      {/* Grid */}
      <View style={{ flexDirection: "row" }}>
        {data.cols.map((week, ci) => (
          <View key={ci} style={{ marginRight: GAP }}>
            {week.map((cell, ri) => (
              <View
                key={ri}
                style={{
                  width: CELL,
                  height: CELL,
                  borderRadius: 3,
                  backgroundColor: LEVEL_COLORS[cell.level],
                  marginBottom: GAP,
                }}
              />
            ))}
          </View>
        ))}
      </View>

      {/* Legend */}
      <View style={{ flexDirection: "row", alignItems: "center", marginTop: 8 }}>
        <Text style={{ color: colors.textMuted, fontSize: 11, marginRight: 6 }}>少</Text>
        {LEVEL_COLORS.map((c, i) => (
          <View key={i} style={{ width: 12, height: 12, borderRadius: 2, backgroundColor: c, marginRight: 3 }} />
        ))}
        <Text style={{ color: colors.textMuted, fontSize: 11, marginLeft: 3 }}>多</Text>
      </View>
    </StayCard>
  );
}
```
