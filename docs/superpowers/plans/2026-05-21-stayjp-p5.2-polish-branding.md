### Task 5: PressableScale — silky press feedback

**Files:**
- Create: `src/components/PressableScale.tsx`

- [ ] **Step 1: Implement using reanimated**

```tsx
// src/components/PressableScale.tsx
import { Pressable, PressableProps, ViewStyle, StyleProp } from "react-native";
import Animated, { useAnimatedStyle, useSharedValue, withSpring } from "react-native-reanimated";
import { ReactNode } from "react";

interface Props extends Omit<PressableProps, "children" | "style"> {
  children: ReactNode;
  style?: StyleProp<ViewStyle>;
  /** Press-down scale (default 0.97). */
  scale?: number;
}

export function PressableScale({ children, style, scale = 0.97, onPress, ...rest }: Props) {
  const s = useSharedValue(1);
  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ scale: s.value }],
  }));
  return (
    <Pressable
      onPressIn={() => { s.value = withSpring(scale, { mass: 0.5, damping: 12, stiffness: 220 }); }}
      onPressOut={() => { s.value = withSpring(1, { mass: 0.5, damping: 12, stiffness: 220 }); }}
      onPress={onPress}
      {...rest}
    >
      <Animated.View style={[animatedStyle, style]}>{children}</Animated.View>
    </Pressable>
  );
}
```
